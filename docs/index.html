/quickstart

CLI in 30 Seconds

terminal — colab-cli 
# 1. Provision GPU VM
colab new --gpu T4 -s my-training

# 2. Upload your script
colab upload train.py /content/train.py

# 3. Launch as detached background job
colab exec -f launch.py --timeout 120

# 4. Monitor — fetch logs, metrics, plots
./fetch.sh                         # download everything
colab download /content/output/model.pt ./

# 5. Tear down
colab stop -s my-training
terminal — multi-account 
# 4 Colab accounts, isolated via $HOME
colab new --gpu T4 -s run1     # hackxie1998 (default)
cb new --gpu T4 -s run2        # stefaniehu929
cc new --gpu T4 -s run3        # xbetterdetermine
clb new --gpu T4 -s run4       # xieminghack
terminal — cron fetch (every 2 min) 
# fetch.sh — downloads log, metrics.json, progress.png
export HTTPS_PROXY=http://127.0.0.1:7890
export ALL_PROXY=socks5://127.0.0.1:7890

colab download -s td3 /content/td3-output/train.log ./output/
colab download -s td3 /content/td3-output/metrics.json ./output/
colab download -s td3 /content/td3-output/plots/progress.png ./output/plots/

# Result: real-time training dashboard, no browser needed
